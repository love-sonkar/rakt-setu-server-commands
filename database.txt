// create folder for postgres

mkdir -p ~/postgres-data

// install postgres

docker pull postgres

// run
docker run --name rakt_setu_db -e POSTGRES_PASSWORD=admin -p 5433:5432 -d postgres

// setup user

docker exec -it rakt_setu_db bash
su - postgres
psql

// create user set hard password

CREATE ROLE admin with login SUPERUSER PASSWORD ‘admin’;

ALTER USER admin WITH CREATEDB CREATEROLE;

// final run command after user setup without volume

docker run --name rakt_setu_db -e POSTGRES_PASSWORD=admin -e POSTGRES_USER=admin -p 5433:5432 -d postgres

// setup volume to store persisted data

docker volume create postgres-volume

docker run --name rakt_setu_db -e POSTGRES_PASSWORD=admin -e POSTGRES_USER=admin -p 5433:5432 -v ~/postgres-data:/var/lib/postgresql -d postgres

// if timeZone error comes fix with this, "-U admin" is username and "-D admin" is db name

docker exec -it <container_name> psql -U admin -d admin

SHOW timezone;
ALTER DATABASE rakt SET timezone TO 'Asia/Kolkata';

// and in intellij run config add vm option
-Duser.timezone=Asia/Kolkata

docker restart <container_name>

// volume in docker

docker volume create postgres-volume

docker run --name rakt_setu_db -e POSTGRES_PASSWORD=admin -e POSTGRES_USER=admin -p 5433:5432 -v ~/postgres-data:/var/lib/postgresql -d postgres
